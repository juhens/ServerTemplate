enum ServerResult : uint16 {
    Success,
    Failed,

    VersionMismatch,
    ServerMaintenance,

    DuplicateAuth,
    TryAgainLater,
    AdultOnly,
    Banned,
    UnknownError,
}

enum PlatformType : uint8 {
    Windows,
    Android,
    IOS,
}

@packet : client
struct C_HandshakeSyn {
    int32 ClientVersion;
    string DeviceId;
    PlatformType PlatformType;
}

@packet : server
struct S_HandshakeSynAck {
    date ServerTime;
    int64 EncryptionSeed;
    ServerResult ServerResult;
}
